module: ietf-keychain
   +--rw keychain
      +--rw private-keys
      |  +--rw private-key* [name]
      |     +--rw name            string
      |     +--ro algorithm?      enumeration
      |     +--ro key-length?     uint32
      |     +--ro public-key?     string
      |     +--rw certificates
      |        +--rw certificate* [name]
      |           +--rw name     string
      |           +--rw chain?   binary
      +--rw trusted-certificates* [name]
         +--rw name                   string
         +--rw trusted-certificate* [name]
            +--rw name           string
            +--rw certificate?   binary
rpcs:
   +---x generate-certificate-signing-request    
   |  +---w input     
   |  |  +---w private-key?   -> /keychain/private-keys/private-key/name
   |  |  +---w subject        binary
   |  |  +---w attributes?    binary
   |  +--ro output    
   |     +--ro certificate-signing-request    binary
   +---x generate-private-key                    
      +---w input     
         +---w name          string
         +---w algorithm     enumeration
         +---w key-length    uint32
